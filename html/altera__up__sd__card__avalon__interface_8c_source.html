01539 short int alt_up_sd_card_fopen(char *name, bool create)
01540 /* This function reads the SD card data in an effort to determine if the card is formated as a FAT16
01541  * volume. Please note that FAT12 has a similar format, but will not be supported by this driver.
01542  * 
01543  * Inputs:
01544  *      name - a file name including a directory, relative to the root directory
01545  *      create - a flag set to true to create a file if it does not already exist
01546  * Output:
01547  *      An index to the file record assigned to the specified file. -1 is returned if the file could not be opened.
01548  *              Return -2 if the specified file has already been opened previously.
01549  */
01550 {
01551         short int file_record_index = -1;
01552 
01553         if ((alt_up_sd_card_is_Present()) && (is_sd_card_formated_as_FAT16))
01554         {
01555         unsigned int home_directory_cluster = 0;
01556         t_file_record home_dir;
01557         
01558         /* First check the file name format. It should not be longer than 12 characters, including a period and the extension.
01559          * Rules:
01560          *  - no spaces
01561          *  - at most 12 chatacters per name, with a period in 9th position.
01562          *  - a / or a \ every at most 12 characters.
01563          */
01564         filename_to_upper_case(name);
01565         if (check_file_name_for_FAT16_compliance(name))
01566         {
01567                         int index;
01568 
01569             /* Get home directory cluster location for the specified file. 0 means root directory. */
01570             if (!get_home_directory_cluster_for_file(name, &home_directory_cluster, &home_dir))

01574             
01575                 /* Find a free file slot to store file specs in. */
01576                 for (index = 0; index < MAX_FILES_OPENED; index++)
01577                 {
01578                         if (active_files[index].in_use == false)
01579                         {
01580                                 file_record_index = index;
01581                                 break;
01582                         }
01583                 }
01584                 if (file_record_index >= 0)
01585                 {
01586                         /* If file record is found, then look for the specified file. If the create flag is set to true 
01587                          * and the file is not found, then it should be created in the current directory. 
01588                          */
01589                 
01590                 if (find_file_in_directory(home_directory_cluster, name, &(active_files[file_record_index])))
01591                 {
01592                     if (create)
01593                     {
01594                         /* Do not allow overwriting existing files for now. */
01595                         return -1;
