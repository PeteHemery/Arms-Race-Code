00034 /*
00036 //
00037 // ALTERA_AVALON_LCD_16207 PERIPHERAL
00038 //
00039 // Provides a hardware interface that allows software to 
00040 // access the two (2) internal 8-bit registers in an Optrex 
00041 // model 16207 (or equivalent) character LCD display (the kind
00042 // shipped with the Nios Development Kit, 2 rows x 16 columns).
00043 //
00044 // Because the interface to the LCD module is "not quite Avalon," 
00045 // the hardware in this module ends-up mapping the module's 
00046 // two physical read-write registers into four Avalon-visible
00047 // registers:  Two read-only registers and two write-only registers.
00048 // A picture is worth a thousand words:
00049 //
00050 // THE REGISTER MAP
00051 // 
00052 //              7     6     5     4     3     2     1     0     Offset
00053 //           +-----+-----+-----+-----+-----+-----+-----+-----+
00054 // RS = 0    |         Command Register (WRITE-Only)         |  0
00055 //           +-----+-----+-----+-----+-----+-----+-----+-----+
00056 // RS = 0    |         Status Register  (READ -Only)         |  1
00057 //           +-----+-----+-----+-----+-----+-----+-----+-----+
00058 // RS = 1    |         Data Register    (WRITE-Only)         |  2
00059 //           +-----+-----+-----+-----+-----+-----+-----+-----+
00060 // RS = 1    |         Data Register    (READ -Only)         |  3
00061 //           +-----+-----+-----+-----+-----+-----+-----+-----+
00062 //
00064 */
