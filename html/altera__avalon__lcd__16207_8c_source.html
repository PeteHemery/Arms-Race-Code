00033 /*
00034  * This file provides the implementation of the functions used to drive a
00035  * LCD panel.
00036  *
00037  * Characters written to the device will appear on the LCD panel as though
00038  * it is a very small terminal.  If the lines written to the terminal are
00039  * longer than the number of characters on the terminal then it will scroll
00040  * the lines of text automatically to display them all.
00041  *
00042  * If more lines are written than will fit on the terminal then it will scroll
00043  * when characters are written to the line "below" the last displayed one - 
00044  * the cursor is allowed to sit below the visible area of the screen providing
00045  * that this line is entirely blank.
00046  *
00047  * The following control sequences may be used to move around and do useful
00048  * stuff:
00049  *    CR    Moves back to the start of the current line
00050  *    LF    Moves down a line and back to the start
00051  *    BS    Moves back a character without erasing
00052  *    ESC   Starts a VT100 style escape sequence
00053  *
00054  * The following escape sequences are recognised:
00055  *    ESC [ <row> ; <col> H   Move to row and column specified (positions are
00056  *                            counted from the top left which is 1;1)
00057  *    ESC [ K                 Clear from current position to end of line
00058  *    ESC [ 2 J               Clear screen and go to top left
00059  *
00060  */

00146 void lcd_write_command(altera_avalon_lcd_16207_state* sp, 
00147   unsigned char command)
00148 {
00149   unsigned int base = sp->base;
00150 
00151   /* We impose a timeout on the driver in case the LCD panel isn't connected.
00152    * The first time we call this function the timeout is approx 25ms 
00153    * (assuming 5 cycles per loop and a 200MHz clock).  Obviously systems
00154    * with slower clocks, or debug builds, or slower memory will take longer.
00155    */
00156   int i = 1000000;
00157 
00158   /* Don't bother if the LCD panel didn't work before */
00159   if (sp->broken)
00160     return;
00161 
00162   /* Wait until LCD isn't busy. */
00163   while (IORD_ALTERA_AVALON_LCD_16207_STATUS(base) & ALTERA_AVALON_LCD_16207_STATUS_BUSY_MSK)
00164     if (--i == 0)
00165     {
00166       sp->broken = 1;
00167       return;
00168     }
00169 
00170   /* Despite what it says in the datasheet, the LCD isn't ready to accept
00171    * a write immediately after it returns BUSY=0.  Wait for 100us more.
00172    */
00173   usleep(100);
00174 
00175   IOWR_ALTERA_AVALON_LCD_16207_COMMAND(base, command);

00236 void lcd_repaint_screen(altera_avalon_lcd_16207_state* sp)
00237 {
00238   int y, x;
00239 
00240   /* scrollpos controls how much the lines have scrolled round.  The speed
00241    * each line scrolls at is controlled by its speed variable - while
00242    * scrolline lines will wrap at the position set by width
00243    */
00244 
00245   int scrollpos = sp->scrollpos;
00246 
00247   for (y = 0 ; y < ALT_LCD_HEIGHT ; y++)
00248   {
00249     int width  = sp->line[y].width;
00250     int offset = (scrollpos * sp->line[y].speed) >> 8;
00251     if (offset >= width)
00252       offset = 0;
00253     
00254     for (x = 0 ; x < ALT_LCD_WIDTH ; x++)
00255     {
00256       char c = sp->line[y].data[(x + offset) % width];
00257 
00258       /* Writing data takes 40us, so don't do it unless required */
00259       if (sp->line[y].visible[x] != c)
00260       {
00261         unsigned char address = x + colstart[y];
00262 
00263         if (address != sp->address)
00264         {
00265           lcd_write_command(sp, LCD_CMD_WRITE_DATA | address);
00266           sp->address = address;
00267         }
00268         lcd_write_data(sp, c);
00269         sp->line[y].visible[x] = c;
00270       }
00271     }
00272   }

00294 void lcd_handle_escape(altera_avalon_lcd_16207_state* sp, char c)
00295 {
00296   int parm1 = 0, parm2 = 0;
00297 
00298   if (sp->escape[0] == '[')
00299   {
00300     char * ptr = sp->escape+1;
00301     while (isdigit(*ptr))
00302       parm1 = (parm1 * 10) + (*ptr++ - '0');
00303 
00304     if (*ptr == ';')
00305     {
00306       ptr++;
00307       while (isdigit(*ptr))
00308         parm2 = (parm2 * 10) + (*ptr++ - '0');
00309     }
00310   }
00311   else
00312     parm1 = -1;
00313 
00314   switch (c)
00315   {
00316   case 'H': /* ESC '[' <y> ';' <x> 'H'  : Move cursor to location */
00317   case 'f': /* Same as above */
00318     if (parm2 > 0)
00319       sp->x = parm2 - 1;
00320     if (parm1 > 0)
00321     {
00322       sp->y = parm1 - 1;
00323       if (sp->y > ALT_LCD_HEIGHT * 2)
00324         sp->y = ALT_LCD_HEIGHT * 2;
00325       while (sp->y > ALT_LCD_HEIGHT)
00326         lcd_scroll_up(sp);
00327     }
00328     break;
00329 
00330   case 'J':
00331     /*   ESC J      is clear to beginning of line    [unimplemented]
00332      *   ESC [ 0 J  is clear to bottom of screen     [unimplemented]
00333      *   ESC [ 1 J  is clear to beginning of screen  [unimplemented]
00334      *   ESC [ 2 J  is clear screen
00335      */
00336     if (parm1 == 2)
00337       lcd_clear_screen(sp);
00338     break;
00339 
00340   case 'K':
00341     /*   ESC K      is clear to end of line
00342      *   ESC [ 0 K  is clear to end of line
00343      *   ESC [ 1 K  is clear to beginning of line    [unimplemented]
00344      *   ESC [ 2 K  is clear line                    [unimplemented]
00345      */
00346     if (parm1 < 1)
00347     {
00348       if (sp->x < ALT_LCD_VIRTUAL_WIDTH)
00349         memset(sp->line[sp->y].data + sp->x, ' ', ALT_LCD_VIRTUAL_WIDTH - sp->x);
00350     }
00351     break;
00352   }
00353 }

00357 int altera_avalon_lcd_16207_write(altera_avalon_lcd_16207_state* sp, 
00358   const char* ptr, int len, int flags)
00359 {
00360   const char* end = ptr + len;
00361 
00362   int y;
00363   int widthmax;
00364 
00365   /* When running in a multi threaded environment, obtain the "write_lock"
00366    * semaphore. This ensures that writing to the device is thread-safe.
00367    */
00368 
00369   ALT_SEM_PEND (sp->write_lock, 0);
00370 
00371   /* Tell the routine which is called off the timer interrupt that the
00372    * foreground routines are active so it must not repaint the display. */
00373   sp->active = 1;
00374 
00375   for ( ; ptr < end ; ptr++)
00376   {
00377     char c = *ptr;
00378 
00379     if (sp->esccount >= 0)
00380     {
00381       unsigned int esccount = sp->esccount;
00382 
00383       /* Single character escape sequences can end with any character
00384        * Multi character escape sequences start with '[' and contain
00385        * digits and semicolons before terminating
00386        */
00387       if ((esccount == 0 && c != '[') ||
00388           (esccount > 0 && !isdigit(c) && c != ';'))
00389       {
00390         sp->escape[esccount] = 0;
00391 
00392         lcd_handle_escape(sp, c);
00393 
00394         sp->esccount = -1;
00395       }
00396       else if (sp->esccount < sizeof(sp->escape)-1)
00397       {
00398         sp->escape[esccount] = c;
00399         sp->esccount++;
00400       }
00401     }
00402     else if (c == 27) /* ESC */
00403     {
00404       sp->esccount = 0;
00405     }
00406     else if (c == '\r')
00407     {
00408       sp->x = 0;
00409     }
00410     else if (c == '\n')
00411     {
00412       sp->x = 0;
00413       sp->y++;
00414 
00415       /* Let the cursor sit at X=0, Y=HEIGHT without scrolling so the user
00416        * can print two lines of data without losing one.
00417        */
00418       if (sp->y > ALT_LCD_HEIGHT)
00419         lcd_scroll_up(sp);
00420     }
00421     else if (c == '\b')
00422     {
00423       if (sp->x > 0)
00424         sp->x--;
00425     }
00426     else if (isprint(c))
00427     {
00428       /* If we didn't scroll on the last linefeed then we might need to do
00429        * it now. */
00430       if (sp->y >= ALT_LCD_HEIGHT)
00431         lcd_scroll_up(sp);
00432 
00433       if (sp->x < ALT_LCD_VIRTUAL_WIDTH)
00434         sp->line[sp->y].data[sp->x] = c;
00435 
00436       sp->x++;
00437     }
00438   }
00439 
00440   /* Recalculate the scrolling parameters */
00441   widthmax = ALT_LCD_WIDTH;
00442   for (y = 0 ; y < ALT_LCD_HEIGHT ; y++)
00443   {
00444     int width;
00445     for (width = ALT_LCD_VIRTUAL_WIDTH ; width > 0 ; width--)
00446       if (sp->line[y].data[width-1] != ' ')
00447         break;
00448 
00449     /* The minimum width is the size of the LCD panel.  If the real width
00450      * is long enough to require scrolling then add an extra space so the
00451      * end of the message doesn't run into the beginning of it.
00452      */
00453     if (width <= ALT_LCD_WIDTH)
00454       width = ALT_LCD_WIDTH;
00455     else
00456       width++;
00457 
00458     sp->line[y].width = width;
00459     if (widthmax < width)
00460       widthmax = width;
00461     sp->line[y].speed = 0; /* By default lines don't scroll */
00462   }
00463 
00464   if (widthmax <= ALT_LCD_WIDTH)
00465     sp->scrollmax = 0;
00466   else
00467   {
00468     widthmax *= 2;
00469     sp->scrollmax = widthmax;
00470 
00471     /* Now calculate how fast each of the other lines should go */
00472     for (y = 0 ; y < ALT_LCD_HEIGHT ; y++)
00473       if (sp->line[y].width > ALT_LCD_WIDTH)
00474       {
00475         /* You have three options for how to make the display scroll, chosen
00476          * using the preprocessor directives below
00477          */
00478 #if 1
00479         /* This option makes all the lines scroll round at different speeds
00480          * which are chosen so that all the scrolls finish at the same time.
00481          */
00482         sp->line[y].speed = 256 * sp->line[y].width / widthmax;
00483 #elif 1
00484         /* This option pads the shorter lines with spaces so that they all
00485          * scroll together.
00486          */
00487         sp->line[y].width = widthmax / 2;
00488         sp->line[y].speed = 256/2;
00489 #else
00490         /* This option makes the shorter lines stop after they have rotated
00491          * and waits for the longer lines to catch up
00492          */
00493         sp->line[y].speed = 256/2;
00494 #endif
00495       }
00496   }
00497 
00498   /* Repaint once, then check whether there has been a missed repaint
00499    * (because active was set when the timer interrupt occurred).  If there
00500    * has been a missed repaint then paint again.  And again.  etc.
00501    */
00502   for ( ; ; )
00503   {
00504     int old_scrollpos = sp->scrollpos;
00505 
00506     lcd_repaint_screen(sp);
00507 
00508     /* Let the timer routines repaint the display again */
00509     sp->active = 0;
00510 
00511     /* Have the timer routines tried to scroll while we were painting?
00512      * If not then we can exit */
00513     if (sp->scrollpos == old_scrollpos)
00514       break;
00515 
00516     /* We need to repaint again since the display scrolled while we were
00517      * painting last time */
00518     sp->active = 1;
00519   }
00520 
00521   /* Now that access to the display is complete, release the write
00522    * semaphore so that other threads can access the buffer.
00523    */
00524 
00525   ALT_SEM_POST (sp->write_lock);
00526 
00527   return len;
00528 }

00591    * the BUSY bit in the status register doesn't work until the display

00598 
00599   /* Wait for another 4.1ms and reset again */
00600   usleep(4100);  
00601   IOWR_ALTERA_AVALON_LCD_16207_COMMAND(base, LCD_CMD_FUNCTION_SET | LCD_CMD_8BIT);
00602 
00603   /* Wait a further 1 ms and reset a third time */
00604   usleep(1000);
00605   IOWR_ALTERA_AVALON_LCD_16207_COMMAND(base, LCD_CMD_FUNCTION_SET | LCD_CMD_8BIT);
00606 
00607   /* Setup interface parameters: 8 bit bus, 2 rows, 5x7 font */
00608   lcd_write_command(sp, LCD_CMD_FUNCTION_SET | LCD_CMD_8BIT | LCD_CMD_TWO_LINE);
00609

00611   lcd_write_command(sp, LCD_CMD_ONOFF);
00612 
00613   /* Clear display */

00617   lcd_write_command(sp, LCD_CMD_MODES | LCD_CMD_MODE_INC);
