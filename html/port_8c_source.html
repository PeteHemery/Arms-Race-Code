00165 /*
00166  * Setup the systick timer to generate the tick interrupts at the required
00167  * frequency.
00168  */
00169 void prvSetupTimerInterrupt( void )
00170 {
00171         /* Try to register the interrupt handler. */
00172         if ( -EINVAL == alt_irq_register( TIMER_IRQ, 0x0, vPortSysTickHandler ) )
00173         { 
00174                 /* Failed to install the Interrupt Handler. */
00175                 asm( "break" );
00176         }
00177         else
00178         {
00179                 /* Configure SysTick to interrupt at the requested rate. */
00180                 IOWR_ALTERA_AVALON_TIMER_CONTROL( TIMER_BASE, ALTERA_AVALON_TIMER_CONTROL_STOP_MSK );
00181                 IOWR_ALTERA_AVALON_TIMER_PERIODL( TIMER_BASE, ( configCPU_CLOCK_HZ / configTICK_RATE_HZ ) & 0xFFFF );
00182                 IOWR_ALTERA_AVALON_TIMER_PERIODH( TIMER_BASE, ( configCPU_CLOCK_HZ / configTICK_RATE_HZ ) >> 16 );
00183                 IOWR_ALTERA_AVALON_TIMER_CONTROL( TIMER_BASE, ALTERA_AVALON_TIMER_CONTROL_CONT_MSK | ALTERA_AVALON_TIMER_CONTROL_START_MSK | ALTERA_AVALON_TIMER_CONTROL_ITO_MSK );     
00184         } 
00185 
00186         /* Clear any already pending interrupts generated by the Timer. */
00187         IOWR_ALTERA_AVALON_TIMER_STATUS( TIMER_BASE, ~ALTERA_AVALON_TIMER_STATUS_TO_MSK );
00188 }

00212 int alt_irq_register( alt_u32 id, void* context, void (*handler)(void*, alt_u32) )
00213 {
00214         int rc = -EINVAL;  
00215         alt_irq_context status;
00216 
00217         if (id < ALT_NIRQ)
00218         {
00219                 /* 
00220                  * interrupts are disabled while the handler tables are updated to ensure
00221                  * that an interrupt doesn't occur while the tables are in an inconsistent
00222                  * state.
00223                  */
00224         
00225                 status = alt_irq_disable_all ();
00226         
00227                 alt_irq[id].handler = handler;
00228                 alt_irq[id].context = context;
00229         
00230                 rc = (handler) ? alt_irq_enable (id): alt_irq_disable (id);
00231         
00232                 /* alt_irq_enable_all(status); This line is removed to prevent the interrupt from being immediately enabled. */
00233         }
