01598 void vTaskPlaceOnEventList( const xList * const pxEventList, portTickType xTicksToWait )
01599 {

01602         /* THIS FUNCTION MUST BE CALLED WITH INTERRUPTS DISABLED OR THE
01603         SCHEDULER SUSPENDED. */
01604 
01605         /* Place the event list item of the TCB in the appropriate event list.
01606         This is placed in the list in priority order so the highest priority task
01607         is the first to be woken by the event. */
01608         vListInsert( ( xList * ) pxEventList, ( xListItem * ) &( pxCurrentTCB->xEventListItem ) );
01609 
01610         /* We must remove ourselves from the ready list before adding ourselves
01611         to the blocked list as the same list item is used for both lists.  We have
01612         exclusive access to the ready lists as the scheduler is locked. */
01613         vListRemove( ( xListItem * ) &( pxCurrentTCB->xGenericListItem ) );
01614 
01615 
01616         #if ( INCLUDE_vTaskSuspend == 1 )
01617         {
01618                 if( xTicksToWait == portMAX_DELAY )
01619                 {
01620                         /* Add ourselves to the suspended task list instead of a delayed task
01621                         list to ensure we are not woken by a timing event.  We will block
01622                         indefinitely. */
01623                         vListInsertEnd( ( xList * ) &xSuspendedTaskList, ( xListItem * ) &( pxCurrentTCB->xGenericListItem ) );
01624                 }
01625                 else
01626                 {
01627                         /* Calculate the time at which the task should be woken if the event does
01628                         not occur.  This may overflow but this doesn't matter. */
01629                         xTimeToWake = xTickCount + xTicksToWait;
01630 
01631                         listSET_LIST_ITEM_VALUE( &( pxCurrentTCB->xGenericListItem ), xTimeToWake );
01632 
01633                         if( xTimeToWake < xTickCount )
01634                         {
01635                                 /* Wake time has overflowed.  Place this item in the overflow list. */
01636                                 vListInsert( ( xList * ) pxOverflowDelayedTaskList, ( xListItem * ) &( pxCurrentTCB->xGenericListItem ) );
01637                         }
01638                         else
01639                         {
01640                                 /* The wake time has not overflowed, so we can use the current block list. */
01641                                 vListInsert( ( xList * ) pxDelayedTaskList, ( xListItem * ) &( pxCurrentTCB->xGenericListItem ) );
01642                         }
01643                 }
01644         }
01645         #else
01646         {
01647                         /* Calculate the time at which the task should be woken if the event does

01650
